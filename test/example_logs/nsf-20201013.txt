. program define calc_weight
  1.     args treatment match_stub n_matches
  2. 
.     forvalues i = 1(1)`n_matches' {
  3.         gen ob = _n // store obs numbers for future use
  4.         quietly save fulldata, replace // need this for remerging, of course
  5. 
.         quietly keep if `treatment' & !missing(`match_stub'`i') // keeping only treatment observations with matches...
  6.         quietly keep `match_stub'`i' // and keep only the var with the obs numbers of the matches
  7.         bysort `match_stub'`i': gen weight`i' = _N // fweight
  8.         quietly by `match_stub'`i': keep if _n == 1 // keep 1 row per control obs
  9. 
.         // merge matched row back to full data, drop _n holder
.         rename `match_stub'`i' ob
 10.         quietly merge 1:m ob using fulldata, nogen
 11.         quietly drop ob
 12.     }
 13. 
.     quietly egen weight = rowtotal(`treatment' weight*) // each use of each row, treatment and matches
 14.     /* tab weight */
. end
